\section*{Definitions and Notation}

Random variable $X \ge 0$, with expectation $E[X]$

Mean: $\mu = E[X]$

Variance: $\sigma^2 = \mathrm{Var}(X)$

Moment Generating Function (MGF): $E[e^{tX}]$

\textbf{Markov‚Äôs Inequality:}
\[
P(X \ge a) \le \frac{E[X]}{a}, \quad a > 0
\]

\textbf{Chebyshev‚Äôs Inequality:}
\[
P(|X - \mu| \ge a) \le \frac{\sigma^2}{a^2}
\]

\textbf{Jensen‚Äôs Inequality (convex $g$):}
\[
g(E[X]) \le E[g(X)]
\]

Convex function condition:
\[
g''(x) > 0
\]

\section*{Assumptions}

\begin{itemize}
\item Random variables are non-negative where required (Markov‚Äôs inequality)
\item Finite mean and variance exist
\item Independence of $X_i$ (in Q-7)
\item Domain constraints: $0 \le Y \le 1$, $X > 0$
\item Equal probability pairing (Q-8)
\end{itemize}

\section*{Proofs / Derivations}

\subsection*{Q-3}

\[
Y = (1 - Y)\cdot 0 + Y \cdot 1
\]

Using convexity:
\[
e^{tY} \le (1 - Y)e^0 + Ye^t
\]

Taking expectation:
\[
E[e^{tY}] \le E[1 - Y + Ye^t]
\]

\[
E[e^{tY}] \le 1 - E[Y] + e^t E[Y]
\]

Let $E[Y] = \mu$:
\[
E[e^{tY}] \le 1 - \mu + \mu e^t
\]

\subsection*{Q-4}

Let $g(x) = x^a$

\[
g'(x) = ax^{a-1}, \quad g''(x) = a(a-1)x^{a-2}
\]

Since $g''(x) > 0$, $g$ is convex.

By Jensen:
\[
(E[X])^a \le E[X^a]
\]

\subsection*{Q-5}

Let $Z = (X - \mu)^2$

\[
P(|X - \mu| \ge k\sigma) = P(Z \ge k^2\sigma^2)
\]

Using Markov:
\[
P(Z \ge k^2\sigma^2) \le \frac{E[Z]}{k^2\sigma^2}
\]

\[
E[(X - \mu)^2] = \sigma^2
\]

\[
P(|X - \mu| \ge k\sigma) \le \frac{1}{k^2}
\]

\subsection*{Q-7}

\[
E[e^{tX_i}] = \frac{e^t + e^{-t}}{2}
\]

Using expansion:
\[
e^t + e^{-t} = 2 \sum_{n=0}^{\infty} \frac{t^{2n}}{(2n)!}
\]

Using inequality:
\[
(2n)! \ge n! 2^n
\]

\[
E[e^{tX_i}] \le e^{t^2/2}
\]

By independence:
\[
E[e^{tS_n}] = \prod E[e^{tX_i}] \le (e^{t^2/2})^n
\]

\subsection*{Q-8}

\[
X = \sum X_i
\]

\[
E[X_i] = \frac{100}{199}
\]

\[
E[X] = 100 \cdot \frac{100}{199}
\]

\[
\mathrm{Var}(X) = \sum \mathrm{Var}(X_i) + 2 \sum \mathrm{Cov}(X_i, X_j)
\]

Using Chebyshev:
\[
P(|X - 50.25| \ge 20.25) \le \frac{25.126}{(20.25)^2}
\]

\[
E[e^{tX}] = \exp(20(e^t - 1))

\[
P(X \ge 26) \le \inf_{t>0} \exp(20(e^t - 1) - 26t)

\[

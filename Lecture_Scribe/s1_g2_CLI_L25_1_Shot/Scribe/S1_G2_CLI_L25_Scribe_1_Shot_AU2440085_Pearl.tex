\section*{1. Problem Setup and Use of Inequalities}

\textbf{Q-1}

Let $X$ denote the annual income of a randomly selected person.\\
Given: 
\[
X \ge 0, \quad E[X] = 40000
\]

\textbf{Markov’s Inequality}

For any $a > 0$,
\[
P(X \ge a) \le \frac{E[X]}{a}
\]

\textbf{(a) Upper Bound Calculation}
\[
P(X \ge 120000) \le \frac{40000}{120000} = \frac{1}{3}
\]

\textbf{(b) Limitation}

Markov’s inequality applies only to probabilities of the form $P(X \ge a)$ for non-negative random variables.

Since $P(X < 10000)$ is a lower-tail probability, Markov’s inequality cannot be applied.

\textbf{Q-2}

Let $X$ denote resistance.\\
Given:
\[
E[X] = \mu = 100, \quad \sigma = 2, \quad \sigma^2 = 4
\]

Defective condition:
\[
|X - \mu| \ge 6
\]

\textbf{Chebyshev’s Inequality}

For any $a > 0$,
\[
P(|X - \mu| \ge a) \le \frac{\sigma^2}{a^2}
\]

\textbf{(a) Probability Bound}
\[
P(|X - 100| \ge 6) \le \frac{4}{36} = \frac{1}{9}
\]

\textbf{(b) Expected Defectives}
\[
10000 \times \frac{1}{9} \approx 1111.11
\]

Hence, approximately 1111 defective resistors.

\section*{2. Jensen’s Inequality and Convexity}

\textbf{Q-3}

Let $0 \le Y \le 1$, $E[Y] = \mu$.\\
Function: $g(x) = e^x$ (convex).

\textbf{Step 1: Convex Combination}
\[
Y = (1 - Y)\cdot 0 + Y \cdot 1
\]

\textbf{Step 2: Convexity Application}
\[
e^{tY} \le (1 - Y)e^0 + Y e^t
\]
\[
e^{tY} \le 1 - Y + Y e^t
\]

\textbf{Step 3: Expectation}
\[
E[e^{tY}] \le E[1 - Y + Y e^t]
\]
\[
E[e^{tY}] \le 1 - E[Y] + e^t E[Y]
\]
\[
E[e^{tY}] \le 1 - \mu + \mu e^t
\]

\textbf{Q-4}

Let $X > 0$, $E[X] = \mu$, define:
\[
g(x) = x^a, \quad a > 1
\]

\textbf{Step 1: Convexity Proof}
\[
g'(x) = ax^{a-1}, \quad g''(x) = a(a-1)x^{a-2}
\]

Since $a > 1$, $x > 0$:
\[
g''(x) > 0 \Rightarrow g(x) \text{ is convex}
\]

\textbf{Step 2: Jensen’s Inequality}
\[
g(E[X]) \le E[g(X)]
\]
\[
(E[X])^a \le E[X^a]
\]

\section*{3. Derivation of Chebyshev’s Inequality}

\textbf{Q-5}

Let $X$ have mean $\mu$, variance $\sigma^2$.

Goal:
\[
P(|X - \mu| \ge k\sigma) \le \frac{1}{k^2}
\]

\textbf{Step 1: Define}
\[
Z = (X - \mu)^2 \ge 0
\]

\textbf{Step 2: Rewrite}
\[
P(|X - \mu| \ge k\sigma) = P((X - \mu)^2 \ge k^2\sigma^2)
\]

\textbf{Step 3: Apply Markov}
\[
P((X - \mu)^2 \ge k^2\sigma^2) \le \frac{E[(X - \mu)^2]}{k^2\sigma^2}
\]

\textbf{Step 4: Use Variance}
\[
E[(X - \mu)^2] = \sigma^2
\]

\[
P(|X - \mu| \ge k\sigma) \le \frac{1}{k^2}
\]

\section*{5. Moment Generating Functions and Bounds}

\textbf{Q-7}

\textbf{Step 1: MGF of $X_i$}

Using expansions:
\[
e^t + e^{-t} = \sum_{n=0}^{\infty} \frac{t^{2n}}{(2n)!}
\]

Using inequality:
\[
(2n)! \ge (n!)^2
\]

\[
\frac{t^{2n}}{(2n)!} \le \frac{(t/2)^{2n}}{n!}
\]

\[
E[e^{tX_i}] \le e^{t^2/2}
\]

\textbf{Step 2: MGF of Sum}
\[
E[e^{tS_n}] = \prod_{i=1}^n E[e^{tX_i}] \le e^{nt^2/2}
\]

\section*{6. Expectation and Variance Computation}

\textbf{Q-8}

Define:
\[
X = \sum_{i=1}^{100} X_i
\]

\textbf{Expectation}
\[
E[X_i] = \frac{i}{100}, \quad E[X] = \frac{100 \cdot 199}{100}
\]

\textbf{Covariance}
\[
E[X_i X_j] = \frac{199}{100} \cdot \frac{197}{99}
\]

\textbf{Variance}
\[
Var(X) = \sum Var(X_i) + \sum_{i<j} Cov(X_i, X_j)
\]

\[
Var(X) \approx 25.126
\]

\textbf{Chebyshev}
\[
P(X \le 30) \le \frac{25.126}{(20.25)^2} \approx 0.061 \approx 0.058
\]

\section*{8. Chernoff Bound Application}

\textbf{Q-10}

Let $X \sim Poisson(20)$.

\[
E[X] = 20, \quad Var(X) = 20
\]

\textbf{(a) Markov}
\[
P(X \ge 26) \le \frac{20}{26}
\]

\textbf{(b) One-sided Chebyshev}
\[
P(X \ge 26) \le \frac{20}{20 + 36}
\]

\textbf{(c) Chernoff Bound}

MGF:
\[
E[e^{tX}] = \exp(20(e^t - 1))
\]

Bound:
\[
P(X \ge 26) \le \inf_{t>0} \exp(20(e^t - 1) - 26t)
\]

Optimize:
\[
20e^t = 26 \Rightarrow t = \ln(1.3)
\]

Final:
\[
P(X \ge 26) \le e^{-26 \ln(1.3)} \approx 0.439
\]

\section*{Final Summary}

Markov: upper tail bounds for non-negative variables\\
Chebyshev: variance-based deviation bounds\\
Jensen: convexity-based expectation inequalities\\
Chernoff: exponential tail bounds via MGFs

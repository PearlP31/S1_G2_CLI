\section{Core Results}

\subsection{Uniform Random Variable}

\subsubsection{Probability Density Function (PDF)}
\[
f_X(x)=
\begin{cases}
\dfrac{1}{b-a}, & a \le x \le b, \\
0, & \text{elsewhere}.
\end{cases}
\]

\subsubsection{Cumulative Distribution Function (CDF)}
\[
F_X(x)=
\begin{cases}
0, & x < a, \\
\dfrac{x-a}{\,b-a\,}, & a \le x < b, \\
1, & x \ge b.
\end{cases}
\]

\subsubsection{Probability Property}
\[
\Pr(a<X<b)=\dfrac{\text{length of interval}}{b-a}
\]

\subsection{Exponential Random Variable}

\subsubsection{Probability Density Function (PDF)}
For $b>0$,
\[
f_X(x)=\frac{1}{b}e^{-x/b}u(x)
\]

\subsubsection{Cumulative Distribution Function (CDF)}
\[
F_X(x)=\left[1-e^{-x/b}\right]u(x)
\]

\section{Analytical Development}

\subsection{Uniform RV Probability Calculation}
\[
\Pr(a<\theta<b)=\int_a^b \frac{1}{2\pi}\, d\theta
\]

\subsection{Conditional Probability Used}
\[
\Pr(A \mid B)=\frac{\Pr(A \cap B)}{\Pr(B)}
\]

\subsection{Trigonometric Condition}
\[
\cos\theta < \frac{1}{2}
\;\Rightarrow\;
\theta \in \left(\frac{\pi}{3},\frac{5\pi}{3}\right)
\]

\subsection{Exponential RV Transformation}
\[
\Pr(3X<y)=\Pr\!\left(X<\frac{y}{3}\right)
\]

\section{Illustrative Applications}

\subsection{Example 1: Uniform Random Variable}

Given:
\[
f_\theta(\theta)=
\begin{cases}
\dfrac{1}{2\pi}, & 0 \le \theta < 2\pi, \\
0, & \text{otherwise}.
\end{cases}
\]

\begin{enumerate}[label=(\alph*)]
    \item
    \[
    \Pr\!\left(\theta>\frac{3\pi}{4}\right)
    =\frac{2\pi-\frac{3\pi}{4}}{2\pi}
    =\frac{5}{8}
    \]

    \item
    \[
    \Pr\!\left(\theta<\pi \mid \theta>\frac{3\pi}{4}\right)
    =\frac{1/8}{5/8}
    =\frac{1}{5}
    \]

    \item
    \[
    \Pr(\cos\theta<1/2)
    =\frac{\frac{5\pi}{3}-\frac{\pi}{3}}{2\pi}
    =\frac{2}{3}
    \]
\end{enumerate}

\subsection{Example 2: Exponential Random Variable}

Given:
\[
f_X(x)=e^{-x}u(x)
\]

\begin{enumerate}[label=(\alph*)]
    \item
    \[
    \Pr(3X<5)
    \]

    \item
    \[
    \Pr(3X<y), \quad \text{for arbitrary } y
    \]
\end{enumerate}

\section{Structural Relationships}

\begin{itemize}
    \item Continuous RV $\rightarrow$ PDF $\rightarrow$ CDF $\rightarrow$ Probability
    \item Uniform RV precedes Exponential RV
    \item Examples follow immediately after definitions
    \item Analytical steps use integration and conditional probability
    \item Applications directly use derived formulas
\end{itemize}

\newpage

\section{Consolidated Summary}

\begin{itemize}
    \item Uniform random variable has constant PDF over a finite interval
    \item Exponential random variable has decaying PDF for $x \ge 0$
    \item PDF integrates to obtain probabilities
    \item CDF used for inequality-based probability queries
    \item Conditional probability formula applied in examples
    \item Trigonometric inequalities converted to interval probabilities
\end{itemize}
